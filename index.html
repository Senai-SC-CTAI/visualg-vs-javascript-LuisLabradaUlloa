Algoritmo 61 em Java script:

let joao, pedro, anos
joao = 175;
pedro = 160;
anos = 0;
while (pedro <= joao) {
joao = joao + 1;
pedro = pedro + 3;
anos = anos + 1;
}
console.log("Pedrinho será mais alto que Joãozinho em", anos, "anos!");
